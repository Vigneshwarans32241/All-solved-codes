import java.util.*;
class Main{
  public static void main(String[] args){
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();
    int m = sc.nextInt();
    int[] arr = new int[m];
    for(int i = 0;i<n;i++){
      int colour = sc.nextInt();
      int val = sc.nextInt();
      if(arr[colour-1]<val) arr[colour-1]=val;
    }
    for(int i = 0;i<m;i++){
      if(arr[i]==0) System.out.print("-1 ");
      else System.out.print(arr[i]+" ");
    }
  }
}
