Requirement: List delivery agents with total number of packages delivered, sorted by highest deliveries first.

Active QuerySELECT a.agent_id, a.name, a.region, COUNT(p.package_id) AS total_delivered FROM agents a LEFT JOIN packages p ON a.agent_id = p.agent_id AND p.status = 'Delivered' GROUP BY a.agent_id, a.name, a.region ORDER BY total_delivered DESC;
